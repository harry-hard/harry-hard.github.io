DLLEXPORT ULONG_PTR WINAPI ReflectiveLoader() {
    // 1. 定位自身 PE 头（略）
    // 2. 分配内存并复制 PE 头和节区（略）
    // 3. 处理重定位
    ProcessRelocations(memBase, delta);
    // 4. 解析导入表
    ResolveImports(memBase);
    // 5. 设置内存权限
    SetMemoryProtections(memBase);
    // 6. 调用 DllMain
    CallDllMain(memBase, DLL_PROCESS_ATTACH);
    return memBase;
}
